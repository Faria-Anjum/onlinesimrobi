def test_selectPrepaidSim(page: Page):

    prepaid.navigate()
    prepaid.findPrepaidSim()
    prepaid.selectPrepaidSimKnowMore()
    prepaid.selectPrepaidSimOrderNow()

def test_selectSimNumber(page: Page):
    sim = RegistrationPage(page)
    sim.navigate()
    sim.findSimNumber()
    sim.selectSimNumber()

def test_fillUpCustomerDetails(page: Page):
    sim = RegistrationPage(page)
    test_selectSimNumber(page)
    sim.fillUpCustomerDetails()
